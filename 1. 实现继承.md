```
function Animal({type}){
  this.type = type
}
Animal.prototype = {
  constructor: Animal,
  eat: ()=>{},  // 哺乳类都有的公有属性
  drink: ()=>{}
}
function Human({name,age}){
  this.name = name  // 人类的私有属性
  this.age = age
}
Human.prototype = {
  constructor: Human,
  love: ()=>{}  // 人类都有的公有属性
}
Human.prototype.__proto__ = Animal.prototype
function Solider(option){
  Animal.call(this,option)  // 继承私有属性
  Human.call(this,option)
  this.ID = option.ID  // 士兵的私有属性
  this.life = option.life
}
Solider.prototype = {  // 士兵的公有属性
  constructor: Solider,
  攻击: ()=>{},
  疾走: ()=>{}
}
Solider.prototype.__proto__ = Human.prototype  // 继承人类的公有属性
var s = new Solider({
  type: '哺乳类',
  name: 'Jack',
  age: 21,
  ID: 100,
  life: 100
})
```
