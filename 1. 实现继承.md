```
class Animal{
  constructor({type}){
    this.type = type
  }
  eat(){}  // 动物的公有属性
  drink(){}
}
class Human extends Animal{
  constructor({name,age,type}){
    super({type})  // 继承动物的私有属性
    this.name = name  // 人类的私有属性
    this.age = age
  }
  eyes = 2  // 非函数的公有属性
  love(){}  // 人类的公有属性
}
class Solider extends Human{
  constructor(option){
    super(option)
    this.life = option.life
    this.ID = option.ID
  }
  疾走(){}
  攻击(){}
}
var s = new Solider({
  type: '哺乳类',
  name: 'Jack',
  age: 21,
  ID: 100,
  life: 100
})
```
