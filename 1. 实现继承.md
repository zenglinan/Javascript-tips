原型链的每一环都可以有公有属性和私有属性，都能被继承。<br>
定义在构造函数里的是私有属性<br>
定义在prototype上的是公有属性
# 基于原型链实现
```
function Animal({type}){
    this.type = type
}
Animal.prototype = {
  constructor: Animal,
  eat(){},  // 动物的公有属性
  drink(){}
}
function Human({name,age,type}){
  this.name = name  // 人类的私有属性
  this.age = age
}
Human.prototype = Object.create(Animal.prototype)
Human.prototype.eyes = 2
Human.prototype.love = function(){}
function Solider(option){
  Animal.call(this,option)  // 初始化Animal上的属性
  Human.call(this,option)   // 初始化Human上的属性
  this.life = option.life
  this.ID = option.ID
}
Solider.prototype = Object.create(Human.prototype)  // 相当于将Solider.prototype.__proto__指向Human.prototype
Solider.prototype.攻击 = function(){}
Solider.prototype.疾走 = function(){}
var s = new Solider({
  type: '哺乳类',
  name: 'Jack',
  age: 21,
  ID: 100,
  life: 100
})

```

# 基于class实现
```
class Animal{
  constructor({type}){
    this.type = type
  }
  eat(){}  // 动物的公有属性
  drink(){}
}
class Human extends Animal{  //声明继承关系，相当于```Human.prototype.__proto__ = Animal.prototype```，也相当于```Human.prototype = Object.create(Animal.prototype)```
  constructor({name,age,type}){
    super({type})  // 初始化并继承动物的私有属性
    this.name = name  // 人类的私有属性
    this.age = age
  }
  eyes = 2  // 非函数的公有属性
  love(){}  // 人类的公有属性
}
class Solider extends Human{
  constructor(option){
    super(option)
    this.life = option.life
    this.ID = option.ID
  }
  疾走(){}
  攻击(){}
}
var s = new Solider({
  type: '哺乳类',
  name: 'Jack',
  age: 21,
  ID: 100,
  life: 100
})

```
